Bootstrapping

If you're using FubuMVC.Validation, bootstrapping is done for you. However, if you're using this outside of FubuMVC you can also install the FubuValidation.StructureMap NuGet. Here's some sample usage of bootstrapping: 

  1. 	[TestFixture]
  2. 	public class StructureMapSmokeTest
  3. 	{
  4.  
  5. 		private IValidator validator
  6. 		{
  7. 			get
  8. 			{
  9. 				var container = new Container();
  10.  
  11. 				ValidationConfiguration.Bootstrap(validation =>
  12. 				{
  13. 					validation.StructureMap(container);
  14.  
  15. 					validation
  16. 						.Registration
  17. 						.AddFromAssemblyContaining<TargetDslRules>();
  18. 				});
  19.  
  20. 				return container.GetInstance<IValidator>();
  21. 			}
  22. 		}
  23.  
  24. 		[Test]
  25. 		public void validates_the_target()
  26. 		{
  27. 			var target = new Target();
  28. 			var notification = validator.Validate(target);
  29.  
  30. 			notification.MessagesFor<Target>(x => x.Name).ShouldHaveCount(1);
  31. 		}
  32.  
  33. 		[Test]
  34. 		public void conditional_rule_from_dsl_registration()
  35. 		{
  36. 			var target = new Target { Name = "TooShort" };
  37. 			var notification = validator.Validate(target);
  38.  
  39. 			notification.MessagesFor<Target>(x => x.Name).ShouldHaveCount(1);
  40. 		}
  41.  
  42. 		public class Target
  43. 		{
  44. 			[Required]
  45. 			public string Name { get; set; }
  46. 		}
  47.  
  48. 		public class TargetDslRules : ClassValidationRules<Target>
  49. 		{
  50. 			public TargetDslRules()
  51. 			{
  52. 				Property(x => x.Name)
  53. 					.MinimumLength(10)
  54. 					.IfValid();
  55. 			}
  56. 		}
  57. 	}

If you're using a different IoC container (and not using FubuMVC, here is the list of services you will need to register:

  1.     public class FubuValidationRegistry : Registry
  2.     {
  3.         public FubuValidationRegistry()
  4.         {
  5.             For<ITypeResolver>().Use<TypeResolver>();
  6. 			For<IServiceLocator>().Use<StructureMapServiceLocator>();
  7.  
  8. 			For<IFieldValidationQuery>().Use<FieldValidationQuery>();
  9. 			For<IValidator>().Use<Validator>();
  10.  
  11.             ForSingletonOf<ITypeDescriptorCache>().Use<TypeDescriptorCache>();
  12.             ForSingletonOf<ValidationGraph>();
  13.             ForSingletonOf<IFieldRulesRegistry>().Use<FieldRulesRegistry>();
  14.             
  15.         }
  16.     }
